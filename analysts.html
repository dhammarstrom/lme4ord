1. Load a few libraries and data sets:

library(lme4ord); library(vegan); data(mite); data(mite.env)

2. Organize the data:

wideData <- dims_to_vars(data.list(mite = as.matrix(mite), mite.env, dimids = c("sites", "species")))
longData <- as.data.frame(wideData)
summary(wideData)

##         mite SubsDens WatrCont Substrate Shrub  Topo sites species
## sites   TRUE     TRUE     TRUE      TRUE  TRUE  TRUE  TRUE   FALSE
## species TRUE    FALSE    FALSE     FALSE FALSE FALSE FALSE    TRUE

3. Fit a simple model with heterogeneous variance function:

formula <- mite ~ scale(WatrCont) + (scale(WatrCont) | species) + multiVarExp(0 + scale(WatrCont))
gm <- strucGlmer(formula, longData, family = poisson(), etastart = log(longData$mite + runif(nrow(longData))))

plot(df$WatrCont, gm$parsedForm$random$NA.multiVarExp$Lambdat$vals,
     las = 1, xlab = "Water content", ylab = "Extra-Poisson variance")
xx <- seq(min(df$WatrCont), max(df$WatrCont), length = 100)
yy <- exp(2 * covarPerTerm(gm)$NA.multiVarExp * (xx - mean(df$WatrCont))/sd(df$WatrCont))
lines(xx, yy)
