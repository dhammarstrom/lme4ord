lme4ord for analysts

Factor models and model-based ordination 

formula <- respVar ~ factAnal(0 + sites | species, nAxes = 2)
strucGlmer(formula, family = binomial(), data = dataFrame)

Heterogeneous variance 

formula <- respVar ~ explVar + (explVar | grpFac) + hetVarExp(explVar)
strucGlmer(formula, family = poisson(), data = dataFrame)

Phylogenetic covariance 

formula <- respVar ~ explVar + (explVar | species) + pagelLambda(1 | species, phylo = tree)
strucGlmer(formula, family = binomial(), data = dataFrame)

Spatial autocorrelation 

formula <- respVar ~ explVar + distCorExp(1 | sites, distMat = dists)
strucGlmer(formula, family = poisson(), data = dataFrame)

Network models 

formula <- respVar ~ explVar + edgeIdent(1 | subjects, edgeMat = adjacencyMat)
strucGlmer(formula, family = poisson(), data = dataFrame)
